def longest_palindromic(a: str) -> str:

    l = len(a)
    res = ""
    for s in range(l):
        if l - s <= len(res):
            break
        for e  in range(s + 1, l + 1):
            subs = a[s:e]
            if subs == subs[::-1] and len(subs)> len(res):
                res = subs

    return res
